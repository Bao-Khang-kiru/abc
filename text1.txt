#include <stdio.h>

void NhapMaTran(int a[100][100], int m, int n, char ten) {
    printf("Nhap ma tran %c:\n", ten);
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            printf("%c[%d][%d] = ", ten, i, j);
            scanf("%d", &a[i][j]);
        }
    }
}

void XuatMaTran(int a[100][100], int m, int n) {
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            printf("%5d", a[i][j]); // %5d để canh lề cho đẹp
        }
        printf("\n");
    }
}

void CongMaTran(int a[100][100], int b[100][100], int c[100][100], int m, int n) {
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            c[i][j] = a[i][j] + b[i][j];
        }
    }
}

int main() {
    int a[100][100], b[100][100], c[100][100];
    int m, n;

    printf("Nhap so dong m: "); scanf("%d", &m);
    printf("Nhap so cot n: "); scanf("%d", &n);

    NhapMaTran(a, m, n, 'A');
    NhapMaTran(b, m, n, 'B');

    CongMaTran(a, b, c, m, n);

    printf("\nMa tran ket qua (A + B):\n");
    XuatMaTran(c, m, n);

    return 0;
}
