#include <stdio.h>

int main() {
    float a[100][100];
    int m, n;
    float tong = 0;

    printf("Nhap so dong: "); scanf("%d", &m);
    printf("Nhap so cot: "); scanf("%d", &n);

    // Nhập ma trận
    printf("Nhap cac phan tu cua ma tran:\n");
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            printf("a[%d][%d] = ", i, j);
            scanf("%f", &a[i][j]);
        }
    }

    // In ma trận để kiểm tra
    printf("\nMa tran vua nhap:\n");
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            printf("%6.2f ", a[i][j]);
        }
        printf("\n");
    }

    // Tính tổng đường chéo chính
    // Đường chéo chính chỉ tồn tại trong phạm vi nhỏ nhất giữa dòng và cột
    int min_size = (m < n) ? m : n; 
    for (int i = 0; i < min_size; i++) {
        tong += a[i][i];
    }

    printf("\nTong cac phan tu tren duong cheo chinh la: %.2f\n", tong);

    return 0;
}
